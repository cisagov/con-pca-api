19def format_params(model_cls=None, params=None): 

20 """ 

21 Format_params. 

22 

23 A utility helper function that takes the filter params dict 

24 and inspects the key names for containing `__` (a double underscore) 

25 as we are using that to denote nested params. It also takes the model class 

26 of the type of model resource that we are applying this params on on it's 

27 ``model_cls`` kwarg and when provided converts params to their native types in python. 

28 

29 :param dict params: 

30 :return params: 

31 :rtype: dict 

32 """ 

33 if params is None: 

34 params = {} 

35 else: 

36 params = params.copy() 

37 for key, value in params.copy().items(): 

38 first, *rest = key.split("__") 

39 if rest: 

40 # currently we only support one level of nestedness 

41 rest = rest[0] 

42 parent_value = params.get(first) 

43 if parent_value: 

44 if not isinstance(parent_value, dict): 

45 params.update({first: {first: parent_value}}) 

46 else: 

47 params[first] = {} 

48 if isinstance(value, str): 

49 try: 

50 value = parse_datetime(value) 

51 except ValueError: 

52 pass 

53 params[first][rest] = value 

54 params.pop(key) 

55 else: 

56 if model_cls: 

57 field = model_cls.fields.get(key) 

58 if field and isinstance(value, str): 

59 if field.typeclass == IntType: 

60 params[key] = int(value) 

61 if field.typeclass == FloatType: 

62 params[key] = float(value) 

63 if field.typeclass == BooleanType: 

64 if value.lower() in ["true", "yes", "1"]: 

65 params[key] = True 

66 elif value.lower() in ["false", "no", "0"]: 

67 params[key] = False 

68 if field.typeclass == DateTimeType: 

69 params[key] = parse_datetime(value) 

70 return params
