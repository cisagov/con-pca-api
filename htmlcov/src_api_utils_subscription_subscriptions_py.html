61def calculate_subscription_start_end_date(start_date): 

62 """Calculates the start and end date for subscription from given start date.""" 

63 now = datetime.now() 

64 

65 if not start_date: 

66 start_date = now.strftime("%Y-%m-%dT%H:%M:%S") 

67 

68 if not isinstance(start_date, datetime): 

69 start_date = datetime.strptime(start_date.split(".")[0], "%Y-%m-%dT%H:%M:%S") 

70 if start_date.replace(tzinfo=None) < now: 

71 start_date = now 

72 

73 start_date = start_date + timedelta(minutes=1) 

74 end_date = start_date + timedelta(minutes=CYCLE_MINUTES) 

75 

76 return start_date, end_date

129def init_subscription_tasks(start_date): 

130 message_types = { 

131 "start_subscription_email": start_date - timedelta(minutes=5), 

132 "monthly_report": start_date + timedelta(minutes=MONTHLY_MINUTES), 

133 "cycle_report": start_date + timedelta(minutes=CYCLE_MINUTES), 

134 "yearly_report": start_date + timedelta(minutes=YEARLY_MINUTES), 

135 "start_new_cycle": start_date + timedelta(minutes=CYCLE_MINUTES), 

136 } 

137 

138 tasks = [] 

139 for message_type, send_date in message_types.items(): 

140 tasks.append( 

141 { 

142 "task_uuid": str(uuid4()), 

143 "message_type": message_type, 

144 "scheduled_date": send_date, 

145 "executed": False, 

146 } 

147 ) 

148 

149 return tasks
